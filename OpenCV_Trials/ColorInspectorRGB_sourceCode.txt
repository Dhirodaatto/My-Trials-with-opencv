
#include <opencv2/imgproc.hpp>
#include <opencv2/highgui.hpp>
#include <iostream>
#include <Windows.h>

using namespace cv;

const int max_color_value = 255;
int red_value = 0;
int green_value = 0;
int blue_value = 0;
const char* win_name = "Color slider";

//Mat final_color;
Mat init_color(600, 800, CV_8UC3, Scalar(0, 0, 0));



void HideConsole()
{
	ShowWindow(GetConsoleWindow(), 0); //0 is the same as SW_HIDE
}
static void redChange(int , void*)
{
	Mat init_color(600, 800, CV_8UC3, Scalar(blue_value, green_value, red_value));
	imshow(win_name, init_color);
}

static void greenChange(int, void*)
{
	Mat init_color(600, 800, CV_8UC3, Scalar(blue_value, green_value, red_value));
	imshow(win_name, init_color);
}

static void blueChange(int, void*)
{
	Mat init_color(600, 800, CV_8UC3, Scalar(blue_value, green_value, red_value));
	imshow(win_name, init_color);
}


int main(int argc, char** argv)
{
	HideConsole();

	if (init_color.empty())
	{
		std::cout << "Something went wrong ... " << std::endl;
		return -1;
	}

	namedWindow(win_name, WINDOW_NORMAL);
	createTrackbar(" Red " , win_name, &red_value, max_color_value, redChange);
	createTrackbar(" Green ", win_name, &green_value, max_color_value, greenChange);
	createTrackbar(" Blue ", win_name, &blue_value, max_color_value, blueChange);
	redChange(0, 0);
	blueChange(0, 0);
	greenChange(0, 0);
	waitKey(0);
	return 0;
}
